Example Races

Introduction

This document provides examples of the mechanics for other subraces.

Dwarf Subraces

Below are the necessary mechanics to model the Mountain Dwarf subrace.

  <thing id="srDwMount" name="Mountain Dwarf" description="Placeholder..." compset="SubRace" uniqueness="unique">
    <fieldval field="rSTR" value="1"/>
    <fieldval field="rHeightMin" value="48"/>
    <fieldval field="rWeightMin" value="130"/>
    <tag group="SubRace" tag="rDwarf"/>
    <tag group="ClVaryName" tag="Replace"/>
    <tag group="HeightDice" tag="2"/>
    <tag group="HeightSize" tag="4"/>
    <tag group="WeightDice" tag="2"/>
    <tag group="WeightSize" tag="6"/>
    <bootstrap thing="raDwfArmTr"/>
    </thing>
  <thing id="raDwfArmTr" name="Dwarven Armor Training" description="Placeholder..." compset="RaceSpec">
    <tag group="ArmProfGrp" tag="ArmorLight"/>
    <tag group="ArmProfGrp" tag="ArmorMed"/>
    <eval phase="First" priority="11000"><![CDATA[
      ~if we've been disabled, get out now
      doneif (tagis[Helper.Disable] <> 0)

      ~assign our weapon proficiencies to the hero
      perform hero.pushtags[ArmProfGrp.Armor?]]]></eval>
    </thing>

Elf Subraces

Below are the necessary mechanics to model the Wood Elf and Dark Elf subraces.

  <thing id="srElfWood" name="Wood Elf" description="Placeholder..." compset="SubRace" uniqueness="unique">
    <fieldval field="rWIS" value="1"/>
    <fieldval field="rHeightMin" value="54"/>
    <fieldval field="rWeightMin" value="100"/>
    <tag group="SubRace" tag="rElf"/>
    <tag group="ClVaryName" tag="Replace"/>
    <tag group="HeightDice" tag="2"/>
    <tag group="HeightSize" tag="10"/>
    <tag group="WeightDice" tag="1"/>
    <tag group="WeightSize" tag="4"/>
    <bootstrap thing="raHiElfWep"/>
    <bootstrap thing="raWdElfFlt"/>
    <bootstrap thing="raWdElfMsk"/>
    </thing>
  <thing id="raWdElfFlt" name="Fleet of Foot" description="Placeholder..." compset="RaceSpec">
    <eval phase="First" priority="750">
      <after name="Base Race Racial Stats"/><![CDATA[
      hero.child[Speed].field[tSpeed].value += 5]]></eval>
    </thing>
  <thing id="raWdElfMsk" name="Mask of the Wild" description="Placeholder..." compset="RaceSpec">
    <tag group="Helper" tag="ShowSpec"/>
    </thing>
  <thing id="srElfDrow" name="Dark Elf (Drow)" description="Placeholder..." compset="SubRace" uniqueness="unique">
    <fieldval field="rCHA" value="1"/>
    <fieldval field="rHeightMin" value="53"/>
    <fieldval field="rWeightMin" value="75"/>
    <tag group="SubRace" tag="rElf"/>
    <tag group="ClVaryName" tag="Replace"/>
    <tag group="SplAttr" tag="aCHA"/>
    <tag group="HeightDice" tag="2"/>
    <tag group="HeightSize" tag="6"/>
    <tag group="WeightDice" tag="1"/>
    <tag group="WeightSize" tag="6"/>
    <bootstrap thing="raDarkVis">
      <autotag group="Value" tag="120"/>
      </bootstrap>
    <bootstrap thing="raDrElfWep"/>
    <bootstrap thing="raDrwSunSe"/>
    <bootstrap thing="spDancLigh">
      <autotag group="Usage" tag="AtWill"/>
      <autotag group="Helper" tag="RaceSpell"/>
      </bootstrap>
    <bootstrap thing="spFaerFire">
      <containerreq phase="First" priority="500"><![CDATA[count:Classes.? >=3]]></containerreq>
      <autotag group="Usage" tag="Day"/>
      <autotag group="Helper" tag="RaceSpell"/>
      <assignval field="trkMax" value="1"/>
      </bootstrap>
    <bootstrap thing="spDarkness">
      <containerreq phase="First" priority="500"><![CDATA[count:Classes.? >= 5]]></containerreq>
      <autotag group="Usage" tag="Day"/>
      <autotag group="Helper" tag="RaceSpell"/>
      <assignval field="trkMax" value="1"/>
      </bootstrap>
    </thing>
  <thing id="raDrElfWep" name="Drow Weapon Training" description="Placeholder..." compset="RaceSpec">
    <tag group="WepProf" tag="wRapier"/>
    <tag group="WepProf" tag="wShortswor"/>
    <tag group="WepProf" tag="wCrssbwHan"/>
    <eval phase="First" priority="11000"><![CDATA[
      ~if we've been disabled, get out now
      doneif (tagis[Helper.Disable] <> 0)

      ~assign our weapon proficiencies to the hero
      perform hero.pushtags[WepProf.?]

      perform hero.pushtags[EasyWeapon.?]]]></eval>
    </thing>
  <thing id="raDrwSunSe" name="Sunlight Sensitivity" description="Placeholder..." compset="RaceSpec">
    <tag group="Helper" tag="ShowSpec"/>
    <eval phase="Final" priority="10000"><![CDATA[
      ~if we've been replaced, get out now
      doneif (tagis[Helper.Disable] <> 0)

      #situational[hero.child[Attack],"Disadvantage when you or target are in bright sunlight",field[thingname].text]
      #situational[hero.child[skPercep],"Disadvantage when you or what you are trying to percieve are in bright sunlight",field[thingname].text]]]></eval>
    </thing>

Halfling Subraces

Below are the necessary mechanics to model the Stout subrace.

  <thing id="srHlnStout" name="Stout" description="Placeholder..." compset="SubRace" uniqueness="unique">
    <fieldval field="rCON" value="1"/>
    <tag group="SubRace" tag="rHalfling"/>
    <tag group="ClVaryName" tag="Prepend"/>
    <bootstrap thing="raStoResil"/>
    </thing>
  <thing id="raStoResil" name="Stout Resilience" description="Placeholder..." compset="RaceSpec">
    <tag group="Helper" tag="ShowSpec"/>
    <tag group="DamageRes" tag="dtPoison"/>
    <eval phase="PreLevel" priority="10000"><![CDATA[
      ~if we've been replaced, get out now
      doneif (tagis[Helper.Disable] <> 0)
      perform hero.pushtags[DamageRes.?]
      #situational[hero.child[svAll],"Advantage vs. poison",field[thingname].text]]]></eval>
    </thing>
