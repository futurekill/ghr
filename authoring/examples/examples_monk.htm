Way of the Four Elements

Below are the necessary mechanics to model the Way of the Four Elements and associated abilities.

  <thing id="cMnkWaFoEl" name="Way of the Four Elements" description="Placeholder..." compset="CustomSpec" uniqueness="unique">
    <tag group="SpecSource" tag="cHelpMnk"/>
    <tag group="abCategory" tag="MnkMonTrad" name="Monk Monastic Tradition"/>
    <tag group="Helper" tag="Primary"/>
    <bootstrap thing="cMnkEleAtt">
      <autotag group="ClSpecWhen" tag="3"/>
      </bootstrap>
    <eval phase="PostLevel" priority="9000"><![CDATA[
       doneif (tagis[Helper.Disable] <> 0)

       linkage[table].field[cSpec2ndNm].text = "Elemental Disciplines"
       linkage[table].field[cSpec2ndSi].text = "Elemental Discipline"

       linkage[table].field[cCustScTot].arrayvalue[2] += 1
       linkage[table].field[cCustScTot].arrayvalue[5] += 2
       linkage[table].field[cCustScTot].arrayvalue[10] += 3
       linkage[table].field[cCustScTot].arrayvalue[16] += 4]]></eval>
    </thing>
  <thing id="cMnkEleAtt" name="Elemental Attunement" description="Placeholder..." compset="ClSpecial" summary="Placeholder...">
    <tag group="abRange" tag="Sight"/>
    <tag group="abDuration" tag="Instant"/>
    <tag group="abAction" tag="Action"/>
    <tag group="SpecSource" tag="cHelpMnk"/>
    <tag group="abCategory" tag="MnkEleDisc" name="Monk Elemental Discipline"/>
    </thing>
